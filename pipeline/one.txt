Activating the virtual environment
source $(pwd)/pipeline/.venv/bin/activate

Running Postgres in a container:
docker run -it --rm \
  -e POSTGRES_USER="root" \
  -e POSTGRES_PASSWORD="root" \
  -e POSTGRES_DB="ny_taxi" \
  -v ny_taxi_postgres_data:/var/lib/postgresql \
  -p 5432:5432 \
  --network=pg-network \
  --name pgdatabase \
  postgres:18

Check....
uv run pgcli -h localhost -p 5432 -u root -d ny_taxi


Running script locally in a virtual env.
uv run ingest_data.py \
  --pg-user=root \
  --pg-pass=root \
  --pg-host=localhost \
  --pg-port=5432 \
  --pg-db=ny_taxi \
  --target-table=yellow_taxi_trips

Dockerizing the ingest script ...
docker run -it \
  --network=pipeline_pg-network \
taxi_ingest:v001 \
  --pg-user=root \
  --pg-pass=root \
  --pg-host=pgdatabase \
  --pg-port=5432 \
  --pg-db=ny_taxi \
  --target-table=yellow_taxi_trips\
  --chunksize=100000
